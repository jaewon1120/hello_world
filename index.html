class Enemy { constructor(x, y, type) { this.x = x; this.y = y; this.type = type; this.toRemove = false; this.isBoss = type.startsWith('BOSS_'); this.abilityTimer = 0; this.bossName = ""; this.shootCooldown = Math.random() * 2000; // 1. 몬스터 타입별 기본 데이터 세팅 if (type === 'FAST') { this.radius = 11; this.speed = 2.6; this.hp = 1; this.color = '#ff6b6b'; this.baseExp = 1; } else if (type === 'SHOOTER') { this.radius = 14; this.speed = 1.2; this.hp = 2; this.color = '#339af0'; this.baseExp = 2; } else if (type === 'BOSS_NECROMANCER') { this.radius = 32; this.speed = 0.8; this.hp = 45; this.color = '#cc5de8'; this.baseExp = 20; this.bossName = bossNames[0]; } else if (type === 'BOSS_TITAN') { this.radius = 36; this.speed = 0.6; this.hp = 70; this.color = '#fcc419'; this.baseExp = 25; this.bossName = bossNames[1]; } else if (type === 'BOSS_HIVEMIND') { this.radius = 34; this.speed = 0.7; this.hp = 55; this.color = '#ff922b'; this.baseExp = 25; this.bossName = bossNames[2]; } else if (type === 'BOSS_BLINK') { this.radius = 35; this.speed = 1.0; this.hp = 60; this.color = '#20c997'; this.baseExp = 30; this.bossName = bossNames[3]; } else if (type === 'BOSS_DOOMSDAY') { this.radius = 38; this.speed = 0.5; this.hp = 85; this.color = '#f03e3e'; this.baseExp = 40; this.bossName = bossNames[4]; } else { this.radius = 15; this.speed = 1.6; this.hp = 3; this.color = '#51cf66'; this.baseExp = 2; } // 2. 보스 격파 수(bossDefeatCount)에 따른 성장 스펙 반영 if (!this.isBoss) { // 일반 몬스터 스펙 업 (이전과 동일) if (type === 'FAST') this.speed += (bossDefeatCount * 0.15); else if (type === 'SHOOTER') this.speed += (bossDefeatCount * 0.08); else this.speed += (bossDefeatCount * 0.12); this.hp = Math.floor(this.hp * (1 + bossDefeatCount * 0.3)); } // 3. ⭐ 경험치 증가 로직 추가 (스펙 성장치 30%보다 작은 15% 비율로 상승) // 최소 1의 경험치는 보장하며, 정수로 떨어지도록 반올림(Math.round) 처리 this.expValue = Math.max(1, Math.round(this.baseExp * (1 + bossDefeatCount * 0.15))); // 최소 체력 보장 if(this.hp < 1) this.hp = 1; this.maxHp = this.hp; } update(dt) { let speedMod = 1.0; if (timeWarpTimer > 0) speedMod = 0.4; const dx = player.x - this.x; const dy = player.y - this.y; const dist = Math.hypot(dx, dy); if (this.type === 'SHOOTER') { if (dist > 250) { this.x += (dx / dist) * this.speed * speedMod; this.y += (dy / dist) * this.speed * speedMod; } else if (dist < 180) { this.x -= (dx / dist) * this.speed * speedMod; this.y -= (dy / dist) * this.speed * speedMod; } this.shootCooldown += dt; if (this.shootCooldown >= 2500) { enemyBullets.push(new EnemyBullet(this.x, this.y, Math.atan2(player.y - this.y, player.x - this.x), 4.5, '#339af0')); this.shootCooldown = 0; } } else { if (dist > 0) { this.x += (dx / dist) * this.speed * speedMod; this.y += (dy / dist) * this.speed * speedMod; } } if (dist < this.radius + player.radius) playerHit(); if (this.isBoss) { this.abilityTimer += dt; if (this.type === 'BOSS_NECROMANCER' && this.abilityTimer >= 4000) { for(let i=0; i<4; i++) enemies.push(new Enemy(this.x + (Math.random()-0.5)*80, this.y + (Math.random()-0.5)*80, 'FAST')); this.abilityTimer = 0; } else if (this.type === 'BOSS_TITAN' && this.abilityTimer >= 2200) { for (let i = 0; i < 8; i++) enemyBullets.push(new EnemyBullet(this.x, this.y, (Math.PI / 4) * i, 3.5, '#fcc419')); this.abilityTimer = 0; } } } draw() { ctx.beginPath(); ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2); ctx.fillStyle = this.color; ctx.fill(); ctx.strokeStyle = '#0b0c10'; ctx.lineWidth = 2; ctx.stroke(); if (this.isBoss) { const barW = this.radius * 2; ctx.fillStyle = 'rgba(0,0,0,0.6)'; ctx.fillRect(this.x - this.radius, this.y - this.radius - 15, barW, 6); ctx.fillStyle = '#ff3e3e'; ctx.fillRect(this.x - this.radius, this.y - this.radius - 15, barW * (this.hp / this.maxHp), 6); } } }
